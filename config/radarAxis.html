# radar

TIP

radar为雷达图提供坐标系支持，如果需要使用雷达图，则需要该配置。

# show

/**
 * @description 是否显示该雷达图坐标系
 * @type {Boolean}
 * @default show = true
 */

# center

/**
 * @description 坐标系中心点
 * @type {Array}
 * @default center = ['50%','50%']
 * @example center = ['50%','50%'] | [100, 100]
 */

# radius

/**
 * @description 坐标系半径
 * @type {String|Number}
 * @default radius = '65%'
 * @example radius = '65%' | 100
 */

# startAngle

/**
 * @description 坐标系起始角度 (弧度值)
 * @type {Number}
 * @default startAngle = -Math.PI / 2
 * @example startAngle = -Math.PI
 */

# axisLabel

/**
 * @description 坐标系标签配置
 * @type {Object}
 */

# axisLabel.show

/**
 * @description 是否显示标签
 * @type {Boolean}
 * @default show = true
 */

# axisLabel.labelGap

/**
 * @description 标签与坐标系间的间距
 * @type {Number}
 * @default labelGap = 15
 */

# axisLabel.color

/**
 * @description 标签颜色 (Hex|rgb|rgba), 将覆盖style.fill
 * @type {Array}
 * @default color = []
 */

# axisLabel.style

/**
 * @description 标签默认样式
 * @type {Object}
 * @default style = {Class Style的配置项}
 */
style: {
  fill: '#333'
}
